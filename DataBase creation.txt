import sqlite3
conn = sqlite3.connect('user_data.db')
cursor = conn.cursor()
cursor.execute('''
    CREATE TABLE IF NOT EXISTS users (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        user_id INTEGER UNIQUE,
        username TEXT,
        role TEXT
    )
''')
cursor.execute('''
    CREATE TABLE IF NOT EXISTS roles (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        role_name TEXT UNIQUE
    )
''')
conn.commit()
cursor.execute("INSERT OR IGNORE INTO roles (role_name) VALUES ('client')")
cursor.execute("INSERT OR IGNORE INTO roles (role_name) VALUES ('admin')")
cursor.execute("INSERT OR IGNORE INTO roles (role_name) VALUES ('support')")
cursor.execute('update users set role = (select role_name from roles where id = 2) where user_id = 768988037')
conn.commit()
